import { Meta, Story } from "@storybook/addon-docs";
import { Imagecredit } from "./Imagecredit";

export const getCaptionForLocale = (locale) => {
  switch (locale) {
    case "english":
      const engText = {
        detail: ["UNDP/Angola", <br />, <strong>Jerome Bell</strong>],
      };
      return engText;
    case "arabic":
      const arabicText = {
        detail: [
          "برنامج الأمم المتحدة الإنمائي / أنغولا",
          <br />,
          <strong>جيروم بيل</strong>,
        ],
      };
      return arabicText;
    case "burmese":
      const burmeseText = {
        detail: ["UNDP/အန်ဂိုလာ", <br />, <strong>ဂျရုမ်း ဘဲလ်</strong>],
      };
      return burmeseText;
    case "japanese":
      const japaneseText = {
        detail: ["UNDP /アンゴラ", <br />, <strong>ジェロームベル</strong>],
      };
      return japaneseText;
    default:
      return { detail: ["UNDP/Angola", <br />, <strong>Jerome Bell</strong>] };
  }
};

# Image Credit Component

Image Credit component used to show the image caption and who posted the image in a related image.

<Story name="Image Credit">
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <Imagecredit label={caption.detail}></Imagecredit>;
  }}
</Story>
