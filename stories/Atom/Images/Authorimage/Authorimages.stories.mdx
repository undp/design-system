export const getCaptionForLocale = (locale) => {
  switch (locale) {
    case "english":
      const engText = {};
      return engText;
    case "arabic":
      const arabicText = {};
      return arabicText;
    case "burmese":
      const burmeseText = {};
      return burmeseText;
    case "japanese":
      const japaneseText = {};
      return japaneseText;
    default:
      return {};
  }
};
