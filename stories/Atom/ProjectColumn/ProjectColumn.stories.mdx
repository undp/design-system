export const getCaptionForLocale = (locale) => {
  switch (locale) {
    case "english":
      const engText = { headertext: "START DATE", subtext: "June 2019" };
      return engText;
    case "arabic":
      const arabicText = { headertext: "تاريخ البدء", subtext: "يونيو 2019" };
      return arabicText;
    case "burmese":
      const burmeseText = {
        headertext: "စတင်သည့်ရက်စွဲ",
        subtext: "ဇွန်လ 2019",
      };
      return burmeseText;
    case "japanese":
      const japaneseText = { headertext: "開始日", subtext: "六月 2019" };
      return japaneseText;
    default:
      return { headertext: "START DATE", subtext: "June 2019" };
  }
};
