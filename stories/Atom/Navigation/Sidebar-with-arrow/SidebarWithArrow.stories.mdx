export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {blockquote: 'Page Title'}; return engText;
    case 'arabic': const arabicText = {blockquote: 'عنوان الصفحة'}; return arabicText;
    case 'burmese': const burmeseText = {blockquote: 'စာမျက်နှာခေါင်းစဉ်'}; return burmeseText;
    case 'japanese': const japaneseText = {blockquote: 'ページタイトル'}; return japaneseText;
    default:
      return {blockquote: 'Page Title'};
  }
};
