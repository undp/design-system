export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {detail: 'menu Item'}; return engText;
    case 'arabic': const arabicText = {detail: 'عنصر القائمة'}; return arabicText;
    case 'burmese': const burmeseText = {detail: 'menu ကို item'}; return burmeseText;
    case 'japanese': const japaneseText = {detail: 'メニュー項目'}; return japaneseText;
    default:
      return {detail: 'menu Item'};
  }
};
