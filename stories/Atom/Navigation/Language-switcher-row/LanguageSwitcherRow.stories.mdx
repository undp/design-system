import { LanguageSwitcherRow } from './LanguageSwitcherRow';

export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {detail: 'Français'}; return engText;
    case 'arabic': const arabicText = {detail: 'الفرنسية'}; return arabicText;
    case 'burmese': const burmeseText = {detail: 'Français'}; return burmeseText;
    case 'japanese': const japaneseText = {detail: 'フランス語'}; return japaneseText;
    default:
      return {detail: 'Français'};
  }
};

# Language Switcher Row Component
Language Switcher Row is used to show the language option in the language switcher component in the header component.

<Story name="Language Row">
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <LanguageSwitcherRow text={caption.detail} ></LanguageSwitcherRow>;
  }}
