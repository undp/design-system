export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {blockquote: 'Adapting to changes in climate and extreme weather events'}; return engText;
    case 'arabic': const arabicText = {blockquote: 'التكيف مع التغيرات المناخية والظواهر الجوية المتطرفة'}; return arabicText;
    case 'burmese': const burmeseText = {blockquote: 'ရာသီဥတုပြောင်းလဲမှုနှင့်အစွန်းရောက်ရာသီဥတုဖြစ်စဉ်များနှင့်လိုက်လျောညီထွေဖြစ်အောင်နေပါ'}; return burmeseText;
    case 'japanese': const japaneseText = {blockquote: '気候変動や異常気象への適応'}; return japaneseText;
    default:
      return {blockquote: 'Adapting to changes in climate and extreme weather events'};
  }
};

The sidebar second-level component is used for showing the nested mapping of the related pages. it shows the second level of the sidebar.
