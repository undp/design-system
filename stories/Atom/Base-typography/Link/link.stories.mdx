export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {detail: 'Default link'}; return engText;
    case 'arabic': const arabicText = {detail: 'الارتباط الافتراضي'}; return arabicText;
    case 'burmese': const burmeseText = {detail: 'မူရင်းလင့်ခ်'}; return burmeseText;
    case 'japanese': const japaneseText = {detail: 'デフォルトのリンク'}; return japaneseText;
    default:
      return {detail: 'Default link'};
  }
};

# Link Component
The Link component defines a hyperlink, which is used to link from one page to another.
<ul>
  <li>We can use the <code>a tag</code> in any component to link a stylesheet or css file to the HTML document.</li>
</ul>
