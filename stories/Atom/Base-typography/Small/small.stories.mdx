export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {detail: 'Et occaecati eos nulla molestiae repellat.'}; return engText;
    case 'arabic': const arabicText = {detail: 'Et Occaecati eos nulla molestiae repellat.'}; return arabicText;
    case 'burmese': const burmeseText = {detail: 'Et occaecati eos nulla molestiae repellat ဖြစ်သည်။'}; return burmeseText;
    case 'japanese': const japaneseText = {detail: 'そして盲目になると、彼は迷惑をかけずに彼らを撃退します。'}; return japaneseText;
    default:
      return {detail: 'Et occaecati eos nulla molestiae repellat.'};
  }
};

# Small Component
The Small component defines smaller text (like copyright and other side-comments).
<ul>
  <li>We can use the <code>small</code> in any component to makes text one font size smaller in the HTML document.</li>
</ul>

Typographic scale when using small in text component:

<Story name="Small">
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <Small label={caption.detail} ></Small>;
  }}
</Story>
