export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {detail: '“Blockquote lorem ipsum dolor sit amet, consectetur adipiscing elit. Pellentesque porttitor mauris quis pulvinar ultrices.”'}; return engText;
    case 'arabic': const arabicText = {detail: '"Blockquote "و ألم الإنترنت نفسه الأريكة الانتقامية'}; return arabicText;
    case 'burmese': const burmeseText = {detail: '“ Blockquote သည်အင်တာနက်ကိုယ်တိုင်၏နာကျင်မှုဖြစ်သည် လက်စားချေရသေးတယ်"'}; return burmeseText;
    case 'japanese': const japaneseText = {detail: '"「Blockquoteはインターネット自体の苦痛です復讐に燃えるソファ"'}; return japaneseText;
    default:
      return {detail: '“Blockquote lorem ipsum dolor sit amet, consectetur adipiscing elit. Pellentesque porttitor mauris quis pulvinar ultrices.”'};
  }
};

# Blockquote Component
The Blockquote component specifies a section that is quoted from another source.
<ul>
  <li>We can use <code>blockquote</code> in any component to render the text found within the <code>blockquote tag</code> as indented text</li>
</ul>

Typographic scale when using blockquote in text component:

<Story name="Blockquote">
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <Blockquote text={caption.detail} ></Blockquote>;
  }}
</Story>
