export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {detail: 'Eos nulla molestiae'}; return engText;
    case 'arabic': const arabicText = {detail: 'لا إزعاج'}; return arabicText;
    case 'burmese': const burmeseText = {detail: 'အဆင်မပြေတာမရှိဘူး'}; return burmeseText;
    case 'japanese': const japaneseText = {detail: '不快感なし'}; return japaneseText;
    default:
      return {detail: 'Eos nulla molestiae'};
  }
};

# Mark Component
The Mark component define the marked text. It is used to highlight the part of the text in a paragraph.
<ul>
  <li>We can use the <code>mark</code> in any component to render the text found within the <code>mark tag</code> as text with a light yellow background color.</li>
</ul>

Typographic scale when using mark in text component:

<Story name="Mark">
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <Mark label={caption.detail} ></Mark>;
  }}
