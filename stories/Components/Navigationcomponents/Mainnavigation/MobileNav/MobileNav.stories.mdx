export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = 'Select Language'; return engText;
    case 'arabic': const arabicText = 'اختار اللغة'; return arabicText;
    case 'burmese': const burmeseText = 'ဘာသာစကားကို ရွေးပါ။'; return burmeseText;
    case 'japanese': const japaneseText = '言語を選択する'; return japaneseText;
    default: return 'Category';
  }
};
export const getLocationDataForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {detail: 'Locations'}; return engText;
    case 'arabic': const arabicText = {detail: 'المواقع'}; return arabicText;
    case 'burmese': const burmeseText = {detail: 'နေရာမျာ'}; return burmeseText;
    case 'japanese': const japaneseText = {detail: '場所'}; return japaneseText;
    default:
      return {detail: 'Locations'};
  }
};
