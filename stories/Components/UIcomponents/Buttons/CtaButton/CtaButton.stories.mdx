export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {detail: 'READ MORE'}; return engText;
    case 'arabic': const arabicText = {detail: 'اقرأ أكثر'}; return arabicText;
    case 'burmese': const burmeseText = {detail: 'ပိုပြီးဖတ်ပါ'}; return burmeseText;
    case 'japanese': const japaneseText = {detail: '続きを読む'}; return japaneseText;
    default:
      return {detail: 'READ MORE'};
  }
};

we are using these modifier classes:
<code>.button</code>, <code>.button--primary</code>, <code>.button--secondary</code>, <code>.button--arrow</code>, <code>.disabled</code>, <code>.cta__link</code>, <code>.cta--arrow</code>

When using button classes on elements that are used to trigger in-page functionality (like collapsing content), rather than linking to new pages or sections within the current page, these links should be given a role="button" to appropriately convey their purpose to assistive technologies such as screen readers.

<div className="subheading">Actions</div>
<ul>
  <li>When an action required by the user has more than one option, always use a negative action button (secondary) paired with a positive action button (primary) in that order.</li>
  <li>Negative action buttons will be on the left.</li>
  <li>Positive action buttons should be on the right.</li>
  <li>When these two types buttons are paired in the correct order, they will automatically space themselves apart.</li>
</ul>
