export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {detail: 'Label'}; return engText;
    case 'arabic': const arabicText = {detail: 'ملصق'}; return arabicText;
    case 'burmese': const burmeseText = {detail: 'တံဆိပ်'}; return burmeseText;
    case 'japanese': const japaneseText = {detail: 'ラベル'}; return japaneseText;
    default:
      return {detail: 'Label'};
  }
};

# Chips
Chips are compact elements that allow users to enter information, select a choice, filter content, or trigger an action.

**Chips:**
Chip with default state, true icon and a cross with allows users to select an option, filter it or add information. Also the cross lets them cancel or delete the information.
