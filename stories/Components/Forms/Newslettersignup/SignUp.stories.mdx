export const getCaptionForLocale = (locale) => {
  switch (locale) {
    case "english":
      const engText = {labelname: "Sign up for our newsletter", buttonname: "SUBMIT", placeholdername: "Email", errorLabel: '*Error: this field is required'};
      return engText;
    case "arabic":
      const arabicText = { labelname: "سجل للحصول على اخر اخبارنا", buttonname: "يقدم", placeholdername: "بريد الالكتروني", errorLabel: '* خطأ: هذا الحقل مطلوب'};
      return arabicText;
    case "burmese":
      const burmeseText = { labelname: "ကျွန်ုပ်တို့၏သတင်းလွှာအတွက်စာရင်းပေးသွင်းပါ", buttonname: "တင်ပေးပါ", placeholdername: "အီးမေးလ်", errorLabel: '*အမှား - ဤအကွက်လိုအပ်သည်'};
      return burmeseText;
    case "japanese":
      const japaneseText = { labelname: "ニュースレターに登録する", buttonname: "送信", placeholdername: "Eメール", errorLabel: '*エラー：このフィールドは必須です'};
      return japaneseText;
    default:
      return { labelname: "Sign up for our newsletter", buttonname: "SUBMIT", placeholdername: "Email", errorLabel: '*Error: this field is required'};
  }
};

# Newsletter Sign Up
The Newsletter signup is a component which users can opt  to receive updates from the organization. 

### Overview
The Newsletter signup components are used by organizations to share important updates, news to their customers.

#### When to use:
- When an option is to be provided to customers to receive updates about the organization..

### Formatting
#### Default
It consists of a title, text bar and chip.

### Behaviors
#### States
There are two states in the Newsletter signup- Default and Hover.
