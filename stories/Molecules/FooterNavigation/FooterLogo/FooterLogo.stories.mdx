export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {headerText: ['United Nations',<br/>,'Development Programme']}; return engText;
    case 'arabic': const arabicText = {headerText: ['الأمم المتحدة',<br/>,'برنامج التطوير']}; return arabicText;
    case 'burmese': const burmeseText = {headerText: ['ကုလသမဂ္ဂ',<br/>,'ဖွံ့ဖြိုးရေးအစီအစဉ်']}; return burmeseText;
    case 'japanese': const japaneseText = {headerText: ['国連',<br/>,'開発プログラム']}; return japaneseText;
    default:
      return {headerText: ['United Nations',<br/>,'Development Programme']};
  }
};

# Footer Logo Component

Footer Logo Component used to display the UNDP or PNUD logo with logo name on the footer section.

<ul>
<li>Two colors variation are created <code>Blue logo with black content</code> and <code>White logo with white content</code>.</li>
</ul>


<Story name="Footer Logo" parameters={{ backgrounds: { default: 'gray'}}}>
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <FooterLogo src={logowhite} logolink="https://www.undp.org" headerText={caption.headerText} style={''} alt={'UNDP Logo'}></FooterLogo>;
  }}
</Story>

<Story name="Footer Logo Inverted" parameters={{ backgrounds: { default: 'gray'}}}>
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <FooterLogo src={logo} logolink="https://www.undp.org" headerText={caption.headerText} style={'inverted'} alt={'UNDP Logo'}></FooterLogo>;
  }}
</Story>
