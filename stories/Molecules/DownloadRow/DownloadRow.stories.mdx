import { Meta, Story } from "@storybook/addon-docs";
import { DownloadRow } from "./DownloadRow";

export const getCaptionForLocale = (locale) => {
  switch (locale) {
    case "english":
      const engText = {
        label: "Title of the publication quite long",
        label2: "PDF (1.2MB)",
        value: "category",
      };
      return engText;
    case "arabic":
      const arabicText = {
        label: "عنوان المنشور طويل جدا",
        label2: "PDF (1.2 ميجابايت)",
        value: "فئة",
      };
      return arabicText;
    case "burmese":
      const burmeseText = {
        label: "စာအုပ်ခေါင်းစဉ်က တော်တော်ရှည်တယ်။",
        label2: "PDF (1.2MB)",
        value: "အမျိုးအစား",
      };
      return burmeseText;
    case "japanese":
      const japaneseText = {
        label: "出版物のタイトルはかなり長い",
        label2: "PDF (1.2MB)",
        value: "カテゴリー",
      };
      return japaneseText;
    default:
      return {
        label: "Title of the publication quite long",
        label2: "PDF (1.2MB)",
        value: "category",
      };
  }
};
