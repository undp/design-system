      const burmeseText = {
        number: "35",
        percentname: "ရာခိုင်နှုန်း",

        number: "35",

    default:
      return {
        number: "35",
        percentname: "Percent",
        text: "Lorem ipsum dolor sit amet, consectetur adipiscing elit.",
      };
  }
};

# Stats Panel Component

## Overview

Stats Panel component defines a card of information data with some items - Image, Number, Percent, and Paragraph. We are using a grid system. It is a multi-usage component that creates boxes that are usually teasing some kind of content.

<Canvas>
  <Story
    name="Stats Panel"
    parameters={{
      docs: {
        source: {
          code: `<div class="grid-x">
  <div class="cell medium-3 statspanel">
    <div class="statspanel__image" style='background-image: url("static/media/stories/assets/images/card2.jpg");'></div>
    <div class="statspanel__box">
      <span class="statspanel__number">35</span>
      <span class="statspanel__per">Percent</span>
      <p>Lorem ipsum dolor sit amet, consectetur adipiscing elit.</p>
    </div>
  </div>
</div>`,
        },
      },
    }}
  >
    {(args, { globals: { locale } }) => {
      const caption = getCaptionForLocale(locale);
      return (
        <StatsPanel
          percent={caption.percentname}
          content={caption.text}
          number={caption.number}
          delaytime='2'
          imageback={BackgroundImg} {...args}
        ></StatsPanel>
      );
    }}
  </Story>
</Canvas>
