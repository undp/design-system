import { Meta, Story, Canvas, Source } from '@storybook/addon-docs';
import { ImageRevelCard } from './ImageCard';
import img from '../../../assets/images/cardimage.jpg';
import imgsmall from '../../../assets/images/CardImgsmall.jpg';

export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = { name: 'Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor incididunt ', btnlabel: 'READ MORE'}; return engText;
    case 'arabic': const arabicText = { name: 'الألم هو الحب نفسه', btnlabel: 'اقرأ أكثر'}; return arabicText;
    case 'burmese': const burmeseText = { name: 'နာကျင်မှုကအချစ်ပါဘဲ', btnlabel: 'ပိုပြီးဖတ်ပါ'}; return burmeseText;
    case 'japanese': const japaneseText = { name: '痛みは愛そのものです', btnlabel: '続きを読む'}; return japaneseText;
    default:
      return { name: 'Lorem ipsum dolor sit amet, consectetur adipiscing elit, sed do eiusmod tempor incididunt ', btnlabel: 'READ MORE'};
  }
};

# Image Reveal Card Component
The Image reveal card component multi usage component which creates boxes that are usually teasing some kind of content like - Image, Text, and CTA Link. We are using the grid system here <code>assets/scss/grid.scss</code>. Atoms tags are used for creating this card.

<div className="subheading">Image Reveal Card- Medium</div>
<p>Image Reveal Card of medium size are shows the Image, and CTA Link. The text will show on hover.</p>

<Story name="Image card">
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <ImageRevelCard contentname={caption.name} image={img} button={caption.btnlabel} link='#' size='medium-4'></ImageRevelCard>;
  }}
</Story>

<div className="subheading">Image Reveal Card- Small</div>
<p>Image Reveal Card of small size comes with a different image, and CTA Link. The text will show on hover.</p>

<Story name="Image card-small">
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <ImageRevelCard contentname={caption.name} image={imgsmall} button={caption.btnlabel} link='#' size='small'></ImageRevelCard>;
  }}
</Story>
