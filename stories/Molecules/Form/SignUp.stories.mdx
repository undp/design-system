import { Meta, Story } from "@storybook/addon-docs";
import { SignUp } from "./SignUp";

export const getCaptionForLocale = (locale) => {
  switch (locale) {
    case "english":
      const engText = {labelname: "Sign up for our newsletter", buttonname: "SUBMIT", placeholdername: "Email", errorLabel: '*Error: this field is required'};
      return engText;
    case "arabic":
      const arabicText = { labelname: "سجل للحصول على اخر اخبارنا", buttonname: "يقدم", placeholdername: "بريد الالكتروني", errorLabel: '* خطأ: هذا الحقل مطلوب'};
      return arabicText;
    case "burmese":
      const burmeseText = { labelname: "ကျွန်ုပ်တို့၏သတင်းလွှာအတွက်စာရင်းပေးသွင်းပါ", buttonname: "တင်ပေးပါ", placeholdername: "အီးမေးလ်", errorLabel: '*အမှား - ဤအကွက်လိုအပ်သည်'};
      return burmeseText;
    case "japanese":
      const japaneseText = { labelname: "ニュースレターに登録する", buttonname: "送信", placeholdername: "Eメール", errorLabel: '*エラー：このフィールドは必須です'};
      return japaneseText;
    default:
      return { labelname: "Sign up for our newsletter", buttonname: "SUBMIT", placeholdername: "Email", errorLabel: '*Error: this field is required'};
  }
};

<Meta title="Molecules/Form/Sign Up"
  argTypes={{
    variant: {
      options: ["black", "white"],
      control: { type: "radio" },
      defaultValue: "black",
    },
  }}
  parameters={{
    backgrounds: {
      default: "white",
      values: [{ name: "gray", value: "#D4D6D8" }],
    },
    previewTabs: {
      'storybook/docs/panel': {
        hidden: true,
      },
    },
  }}
/>

<Story name="Sign Up" parameters={{ backgrounds: { default: "gray" } }}>
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return (
      <SignUp element='input' type='email' required='required' mode='form-email' errorText={caption.errorLabel} label={caption.labelname} button={caption.buttonname} placeholder={caption.placeholdername} {...args}></SignUp>
    );
  }}
</Story>
