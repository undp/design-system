export const getCaptionForLocale = (locale) => {
switch(locale) {
case 'english': const engText = {Startlabel:'START DATE', Start:'June 2019', Endlabel:'END DATE', End:'June 2023', Statuslabel:'STATUS', Status:'Ongoing'}; return engText;
case 'arabic': const arabicText = {Startlabel:'تاريخ البدء', Start:'يونيو 2019', Endlabel:'تاريخ الانتهاء', End:'يونيو 2023', Statuslabel:'حالة', Status:'جاري التنفيذ'}; return arabicText;
case 'burmese': const burmeseText = {Startlabel:'စတင်သည့်ရက်စွဲ', Start:'ဇွန်လ 2019', Endlabel:'ပြီးဆုံးရက်', End:'ဇွန်လ 2023', Statuslabel:'အဆင့်အတန်း', Status:'လုပ်နေဆဲဖြစ်သည်'}; return burmeseText;
case 'japanese': const japaneseText = {Startlabel:'開始日', Start:'六月 2019', Endlabel:'終了日', End:'六月 2023', Statuslabel:'状態', Status:'進行中'}; return japaneseText;
default:
return {Startlabel:'START DATE', Start:'June 2019', Endlabel:'END DATE', End:'June 2023', Statuslabel:'STATUS', Status:'Ongoing'};
}
};

# Project Details Column Component

Project Details Column component defines a section of information data with some items - Start Date, End Date and Status.

<Story name="Project Details Column">
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return (
      <Projectdetails
        startName={caption.Startlabel}
        startText={caption.Start}
        endName={caption.Endlabel}
        endText={caption.End}
        statusName={caption.Statuslabel}
        statusText={caption.Status}
      ></Projectdetails>
    );
  }}
</Story>
