export const getCaptionForLocale = (locale) => {
  switch(locale) {
    case 'english': const engText = {header: 'Our Focus', country: 'The sdgs in action'}; return engText;
    case 'arabic': const arabicText = {header: 'تركيزنا', country: 'أهداف التنمية المستدامة في العمل'}; return arabicText;
    case 'burmese': const burmeseText = {header: 'ကျွန်ုပ်တို့၏အာရုံစူးစိုက်မှု', country: 'SDGS လုပ်ဆောင်ချက်'}; return burmeseText;
    case 'japanese': const japaneseText = {header: '私たちの焦点', country: '動作中のSDGS'}; return japaneseText;
    default:
      return {header: 'Our Focus', country: 'The sdgs in action'};
  }
};

# Page Title Block 

The page Block Component defines a section of information data with some items - Header title, Description. <code>H2</code> <code>H4</code> Atoms tags are used for creating this section.

<Story name="Page Block Black">
  {(args, { globals: { locale } }) => {
    const caption = getCaptionForLocale(locale);
    return <Page headerText={caption.header} countryText={caption.country} style={''}></Page>;
  }}
</Story>
